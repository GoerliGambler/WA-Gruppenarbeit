Funktionen.R       =   "Funktionen-Skript 1" laut Aufgabenstellung

Hilfsfunktionen.R  =   "Funktionen-Skript 2" laut Aufgabenstellung

Auswertung.R       =   "viertes Skript in ihrem Repository" laut Aufgabenstellung
